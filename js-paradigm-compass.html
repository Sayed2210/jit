procedural.js 
// Procedural: step-by-step flow
let scores = [72, 85, 91, 60];
let total = 0;

for (let i = 0; i < scores.length; i++) {
  if (scores[i] >= 70) {
    total = total + scores[i];
  }
}

console.log(`Passing total: ${total}`);
// → Passing total: 248

oop.js 
class Animal {
  constructor(name) {
    this.name = name;
  }
  speak() { return `...`; }
}

class Dog extends Animal {
  speak() {
    return `${this.name} barks!`;
  }
}
const d = new Dog("Rex");
d.speak(); // "Rex barks!"

functional.js 
// Pure, composable, no mutations
const scores = [72, 85, 91, 60];

const result = scores
  .filter(s => s >= 70)    // select
  .map(s => s * 1.1)       // transform
  .reduce((sum, s) =>      // accumulate
    sum + s, 0);

// scores unchanged — immutable!
console.log(result); // 272.8
